<ul class="slider">
  <li class="active">1</li>
  <li>2</li>
  <li>3</li>
</ul>
<div class="prev">
  назад
</div>
<div class="next">
  вперед
</div>

.slider {
  display: flex;
  list-style: none;
}

.slider li {
  padding: 10px;
  background: #ccc;
  margin: 5px;
}

.slider li.active {
  background: red;
}

document.addEventListener("DOMContentLoaded", function(event) { 
  var prev = document.querySelector('.prev');
  var next = document.querySelector('.next');
  
  next.addEventListener("click", function() {
  	var activeLi = document.querySelector('.slider li.active');
    var nextLi = document.querySelector('.slider li.active + li');
    
    if (nextLi) {
    	activeLi.classList.remove("active");
    	nextLi.classList.add("active");
    }
  })
  
  prev.addEventListener("click", function() {
  	var activeLi = document.querySelector('.slider li.active');
    
    var slides = document.querySelectorAll('.slider li');
    var slidesCount = slides ? slides.length : 0;
    
    var activeIndex = 0;
    var prevIndex = 0;
    for (var index = 0; index < slidesCount; index++) {
      if (slides[index] === activeLi) {
        var tempPrevIndex = index - 1;
        if (tempPrevIndex => 0) {
          activeIndex = index;
          prevIndex = tempPrevIndex;
          break;
        }
      }
    }
       
   if (prevIndex !== activeIndex) {
      activeLi.classList.remove("active");
      var prevLi = slides[prevIndex]
      prevLi.classList.add("active");
    }
  })
});
